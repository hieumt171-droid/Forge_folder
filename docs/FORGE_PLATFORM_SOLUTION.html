1. Executive Summary

1.1 Mục tiêu kinh doanh & bài toán cần giải quyết

Tổ chức đang vận hành nhiều Jira Cloud site và cần mở rộng chức năng Jira theo workflow nội bộ mà không mua thêm marketplace app đắt tiền. Nhóm kỹ thuật nhỏ (2–5 dev) cần tự xây dựng và vận hành tooling trực tiếp trong Jira.

1.2 Tóm tắt giải pháp đề xuất

Một Forge App duy nhất — chạy trên Jira Cloud, nhiều site, không host riêng

Xây dựng Atlassian Forge application với các module Jira (Issue Panel, Global Page, Project Page, Admin Page), deploy một lần và install độc lập trên từng Jira site. Toàn bộ backend chạy serverless trên Atlassian infrastructure. Dữ liệu lưu hoàn toàn trong Forge SQL — không cần tự provision database, không KVS optimization trong MVP.

4.2 Data Model & Forge SQL Strategy

Tại sao chỉ dùng Forge SQL trong MVP?

Một nguồn dữ liệu duy nhất — đơn giản để build, đơn giản để bảo vệ.

Forge SQL cho phép SELECT/GROUP BY/aggregate/JOIN — đủ cho mọi tính năng MVP. KVS (key-value store) sẽ xem xét trong Phase 2 nếu profiling cho thấy query timesheet vượt ngưỡng > 3s. Không optimize trước khi có vấn đề thật.

Schema SQL

-- ── Bảng chính: mỗi record = 1 row ──────────────────────────────────────
CREATE TABLE records (
  id            VARCHAR(36)   PRIMARY KEY,           -- uuid v4
  account_id    VARCHAR(128)  NOT NULL,              -- từ context.accountId
  issue_key     VARCHAR(64)   NOT NULL,              -- e.g. HSF-12
  project_key   VARCHAR(32)   NOT NULL,              -- denormalize để query nhanh
  record_type   VARCHAR(64)   NOT NULL,              -- domain-specific category
  duration_min  INT           NOT NULL,              -- đơn vị phút (nếu applicable)
  note          VARCHAR(500),
  logged_at     DATE          NOT NULL,              -- ngày user chọn
  status        ENUM('draft','submitted','approved','rejected') DEFAULT 'draft',
  created_at    DATETIME      DEFAULT CURRENT_TIMESTAMP,
  updated_at    DATETIME      DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_user_week    (account_id, logged_at),
  INDEX idx_project      (project_key, status),
  INDEX idx_issue        (issue_key)
);

-- ── Chống submit trùng tuần (idempotent) ─────────────────────────────────
CREATE TABLE week_submissions (
  account_id   VARCHAR(128)  NOT NULL,
  week_start   DATE          NOT NULL,
  submitted_at DATETIME      DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (account_id, week_start)             -- chỉ 1 row/user/tuần
);

-- ── Cấu hình per-site (thay cho KVS trong MVP) ───────────────────────────
CREATE TABLE site_config (
  config_key   VARCHAR(128)  PRIMARY KEY,
  config_value TEXT,
  updated_by   VARCHAR(128),
  updated_at   DATETIME      DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- ── Phase 2: approval history ─────────────────────────────────────────────
CREATE TABLE approvals (
  id           VARCHAR(36)   PRIMARY KEY,
  record_id    VARCHAR(36)   NOT NULL,
  reviewer_id  VARCHAR(128)  NOT NULL,
  action       ENUM('approved','rejected') NOT NULL,
  comment      VARCHAR(500),
  acted_at     DATETIME      DEFAULT CURRENT_TIMESTAMP,
  INDEX idx_record (record_id)
);

Migration strategy — lazy init khi app khởi động lần đầu

// src/db/client.js
import { sql } from '@forge/sql';
import { storage } from '@forge/api';

const SCHEMA_VERSION = 2; // tăng mỗi khi thêm bảng/column

export const ensureSchema = async () => {
  // Dùng Forge KVS chỉ cho 1 việc duy nhất: track schema version
  const current = await storage.get('db:schema_version') ?? 0;
  if (current >= SCHEMA_VERSION) return;

  if (current < 1) await sql.query(CREATE_TABLE_RECORDS);
  if (current < 1) await sql.query(CREATE_TABLE_WEEK_SUBMISSIONS);
  if (current < 1) await sql.query(CREATE_TABLE_SITE_CONFIG);
  if (current < 2) await sql.query(`ALTER TABLE records ADD COLUMN extra_json TEXT`);

  await storage.set('db:schema_version', SCHEMA_VERSION);
};

// Gọi ở đầu mỗi resolver — nhanh vì chỉ chạy thật khi version cũ
export const withDb = async (fn) => {
  await ensureSchema();
  return fn();
};
Lưu ý về KVS: KVS được dùng duy nhất để track db:schema_version — 1 key, value là số nguyên. Không lưu business data vào KVS trong MVP. Nếu sau này cần cache query chậm, sẽ thêm ở Phase 2 sau khi đo được vấn đề thực tế.

Repository pattern — typed, parameterized

// src/db/repositories/recordRepo.js
import { sql } from '@forge/sql';
import { v4 as uuidv4 } from 'uuid';

export const recordRepo = {
  async insert({ accountId, issueKey, projectKey, recordType, durationMin, note, loggedAt }) {
    const id = uuidv4();
    await sql.query(
      `INSERT INTO records (id, account_id, issue_key, project_key, record_type, duration_min, note, logged_at)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
      [id, accountId, issueKey, projectKey, recordType, durationMin, note ?? null, loggedAt]
    );
    return { id, accountId, issueKey, projectKey, recordType, durationMin, note, loggedAt, status: 'draft' };
  },

  async findByIssueToday(accountId, issueKey, date) {
    const { rows } = await sql.query(
      `SELECT * FROM records WHERE account_id = ? AND issue_key = ? AND logged_at = ? ORDER BY created_at DESC`,
      [accountId, issueKey, date]
    );
    return rows;
  },

  async findByWeek(accountId, weekStart, weekEnd) {
    const { rows } = await sql.query(
      `SELECT issue_key, logged_at, record_type, SUM(duration_min) as total_min
       FROM records
       WHERE account_id = ? AND logged_at BETWEEN ? AND ?
       GROUP BY issue_key, logged_at, record_type
       ORDER BY logged_at`,
      [accountId, weekStart, weekEnd]
    );
    return rows;
  },

  async findById(id) {
    const { rows } = await sql.query(`SELECT * FROM records WHERE id = ?`, [id]);
    return rows[0] ?? null;
  },

  async updateStatus(id, status) {
    await sql.query(`UPDATE records SET status = ? WHERE id = ?`, [status, id]);
  },

  async softDelete(id) {
    await sql.query(`UPDATE records SET status = 'archived' WHERE id = ?`, [id]);
  },
};

submitWeek — idempotent hoàn toàn bằng SQL

defineLogged('submitWeek', async (req) => {
  const accountId = assertSelfOnly(req);
  const { weekStart } = req.payload;
  const weekEnd = addDays(weekStart, 6);

  // Idempotent check: đã submit tuần này chưa?
  const { rows: existing } = await sql.query(
    `SELECT submitted_at FROM week_submissions WHERE account_id = ? AND week_start = ?`,
    [accountId, weekStart]
  );
  if (existing.length > 0) return { ok: true, alreadySubmitted: true, submittedAt: existing[0].submitted_at };

  // Đếm draft trong tuần
  const { rows: drafts } = await sql.query(
    `SELECT COUNT(*) as cnt FROM records WHERE account_id = ? AND logged_at BETWEEN ? AND ? AND status = 'draft'`,
    [accountId, weekStart, weekEnd]
  );
  if (drafts[0].cnt === 0) return { ok: false, reason: 'nothing_to_submit' };

  // Atomic: UPDATE + INSERT (không dùng transaction vì Forge SQL không support multi-statement tx)
  await sql.query(
    `UPDATE records SET status = 'submitted' WHERE account_id = ? AND logged_at BETWEEN ? AND ? AND status = 'draft'`,
    [accountId, weekStart, weekEnd]
  );
  await sql.query(
    `INSERT INTO week_submissions (account_id, week_start) VALUES (?, ?)`,
    [accountId, weekStart]
  );

  return { ok: true, submittedCount: drafts[0].cnt };
});

defineLogged('exportCsv', async (req) => {

  const header = 'date,issue_key,record_type,duration_min,hours,note,status';
  const body = rows.map(r =>
    [r.logged_at, r.issue_key, r.record_type, r.duration_min,
     (r.duration_min / 60).toFixed(2), `"${(r.note ?? '').replace(/"/g, '""')}"`, r.status].join(',')
  ).join('\n');

  return {
    ok: true,
    csvContent: '' + header + '\n' + body,   // BOM để Excel mở đúng UTF-8
    filename: `records-${weekStart}-to-${weekEnd}.csv`,
    rowCount: rows.length,
  };
});

4.5 Permission & Security Model

Quy tắc bất biến: Không bao giờ tin accountId từ UI payload. Luôn dùng context.accountId được Forge Bridge inject — client không thể forge giá trị này. 
// src/resolvers/auth.js

export const assertSelfOnly = (req) => {
  const ctxId = req.context.accountId;
  const payloadId = req.payload?.accountId;
  if (payloadId && payloadId !== ctxId) throw new ForbiddenError('Không được thao tác thay user khác');
  return ctxId;
};

export const assertOwnership = async (recordId, accountId) => {
  const record = await recordRepo.findById(recordId);
  if (!record) throw new NotFoundError('Record không tồn tại');
  if (record.account_id !== accountId) throw new ForbiddenError('Record không thuộc về bạn');
  if (record.status !== 'draft') throw new ForbiddenError('Chỉ sửa/xóa được record ở trạng thái draft');
  return record;
};

export const assertProjectLead = async (projectKey, accountId) => {
  const res = await api.asUser().requestJira(route`/rest/api/3/project/${projectKey}`);
  const project = await res.json();
  if (project.lead?.accountId !== accountId) throw new ForbiddenError('Chỉ project lead mới thực hiện được');
};

export const assertSiteAdmin = async (accountId) => {
  // Kiểm tra Jira site-admin role qua API

4.6 Configuration per Site

Cấu hình per-site được lưu trong bảng site_config (Forge SQL) — đọc/ghi qua resolver, không hardcode trong code. Admin Page (Phase 2) sẽ cung cấp UI để chỉnh.

// src/db/repositories/configRepo.js
export const configRepo = {
  async get(key, defaultValue = null) {
    const { rows } = await sql.query(`SELECT config_value FROM site_config WHERE config_key = ?`, [key]);
    if (!rows.length) return defaultValue;
    try { return JSON.parse(rows[0].config_value); }
    catch { return rows[0].config_value; }
  },

  async set(key, value, updatedBy) {
    const serialized = typeof value === 'string' ? value : JSON.stringify(value);
    await sql.query(
      `INSERT INTO site_config (config_key, config_value, updated_by) VALUES (?, ?, ?)
       ON DUPLICATE KEY UPDATE config_value = VALUES(config_value), updated_by = VALUES(updated_by), updated_at = NOW()`,
      [key, serialized, updatedBy]
    );
  },
};

// src/index.js — defineLogged bọc mọi resolver
const defineLogged = (name, fn) =>
  defineHandler(name, async (req) => {
    const logger = createLogger('platform-resolver');
    const t0 = Date.now();
    logger.start({ action: name, accountId: req.context.accountId });
    try {
      await withDb(() => {});           // ensure schema on first run
      const result = await fn(req);
      logger.success({ action: name, durationMs: Date.now() - t0 });
      return result;
    } catch (err) {

    }
  });
